## Installation

<Steps>

  <Step>
    ### Install the primitive
    
    Install the `@radix-ui/react-tooltip` package.

    ```package-install
    @radix-ui/react-tooltip
    ```

  </Step>
    
  <Step>

## Usage

```ts
import {
  Tooltip,
  TooltipContent,
  TooltipProvider,
  TooltipTrigger,
} from '~/components/ui/tooltip';
```

```tsx
<TooltipProvider>
  <Tooltip>
    <TooltipTrigger>Hover</TooltipTrigger>
    <TooltipContent>
      <p>Add to library</p>
    </TooltipContent>
  </Tooltip>
</TooltipProvider>
```
