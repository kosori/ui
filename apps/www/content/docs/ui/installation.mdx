  <Step>
    ### Add dependencies
    
    Install the following dependencies:
    ```package-install
    clsx next-themes tailwind-merge tailwind-variants tailwindcss-animate
    ```
  </Step>

  <Step>
    ### Add theme provider
    
    Add the `<ThemeProvider />` component at the root of your app.

    <div className='not-prose [&_figure]:max-h-[320px] [&_figure]:overflow-y-auto'>
    ```tsx title="app/layout.tsx"
    import type { Metadata } from 'next';
    import { Inter } from 'next/font/google';
    import './globals.css';
    import { ThemeProvider } from 'next-themes'; // [!code highlight]

    const inter = Inter({ subsets: ['latin'] });

    export const metadata: Metadata = {
      title: 'Create Next App',
      description: 'Generated by create next app',
    };

    export default function RootLayout({
      children,
    }: Readonly<{
      children: React.ReactNode;
    }>) {
      return (
        <html lang='en'>
          <body className={inter.className}>
            <ThemeProvider attribute='class'> // [!code highlight]
              {children}
            </ThemeProvider> // [!code highlight]
          </body>
        </html>
      );
    }
    ```
    </div>

  </Step>

  <Step>
    ### Add dark mode
    
    Set `darkMode` strategy to `class` in your Tailwind CSS config file.
    
    ```ts title="tailwind.config.ts"
    import type { Config } from 'tailwindcss';

    const config: Config = {
      darkMode: 'class',  // [!code highlight]
      // ...
    };
    ```

  </Step>

  <Step>
    ### Add tailwind animations

    Extend the `keyframes` and `animation` objects in your `theme` object.

    ```ts title="tailwind.config.ts"
    const config: Config = {
      darkMode: 'class',
      content: [
        '...',
      ],
      theme: {
        // ...
        extend: {
          // ...
          // [!code highlight:20]
          keyframes: {
            'accordion-down': {
              from: { height: '0' },
              to: { height: 'var(--radix-accordion-content-height)' },
            },
            'accordion-up': {
              from: { height: 'var(--radix-accordion-content-height)' },
              to: { height: '0' },
            },
            'caret-blink': {
              '0%,70%,100%': { opacity: '1' },
              '20%,50%': { opacity: '0' },
            },
          },
          animation: {
            'accordion-down': 'accordion-down 0.2s ease-out',
            'accordion-up': 'accordion-up 0.2s ease-out',
            'caret-blink': 'caret-blink 1.25s ease-out infinite',
          },
        },
      },
    };
    ```

  </Step>

  <Step>
    ### Add tailwind plugins
    
    Add the `tailwindcss-animate` plugin to your Tailwind CSS config.

    ```ts title="tailwind.config.ts"
    import type { Config } from 'tailwindcss';
    import animate from 'tailwindcss-animate'; // [!code highlight]

    const config: Config = {
      // ...,
      plugins: [animate], // [!code highlight]
    };
    ```

  </Step>

    ### Add utilities

    ```ts title="app/utils/cn.ts"
    import type { ClassValue } from 'clsx';
    import { clsx } from 'clsx';
    import { twMerge } from 'tailwind-merge';

    export const cn = (...inputs: ClassValue[]) => twMerge(clsx(inputs));
    ```
