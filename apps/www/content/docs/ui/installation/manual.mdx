<Steps>
  <Step>
    ### Add dependencies
    
    Install the following dependencies:
    ```package-install
    clsx next-themes tailwind-merge tailwind-variants tailwindcss-animate @radix-ui/colors
    ```
  </Step>

  <Step>
    ### Add theme provider
    
    Add the `<ThemeProvider />` component at the root of your app.

    ```tsx title="app/layout.tsx"
    import type { Metadata } from 'next';
    import './globals.css';
    import { ThemeProvider } from 'next-themes'; // [!code highlight]

    export const metadata: Metadata = {
      title: 'Create Next App',
      description: 'Generated by create next app',
    };

    export default function RootLayout({
      children,
    }: Readonly<{
      children: React.ReactNode;
    }>) {
      return (
        <html lang='en'>
          <body>
            // [!code highlight:1]
            <ThemeProvider attribute='class'>
              {children}
            // [!code highlight:1]
            </ThemeProvider>
          </body>
        </html>
      );
    }
    ```

  </Step>

  <Step>
    ### Add dark mode
    
    Set `darkMode` strategy to `class` in your Tailwind CSS config file.
    
    ```ts title="tailwind.config.ts"
    import type { Config } from 'tailwindcss';

    export default {
      darkMode: 'class', // [!code highlight]
      // ...
    } satisfies Config;
    ```

  </Step>
  
  <Step>
    ### Add tailwind border radius
    
    Overwrite the `borderRadius` object in your `theme` object.
    
    ```ts title="tailwind.config.ts"
    import type { Config } from 'tailwindcss';

    export default {
      darkMode: 'class',
      content: [
        '...',
      ],
      theme: {
        // ...
        // [!code highlight:11]
        borderRadius: {
          none: '0px',
          sm: 'calc(var(--radius) - 3px)',
          DEFAULT: 'var(--radius)',
          md: 'var(--radius)',
          lg: 'calc(var(--radius) + 3px)',
          xl: 'calc(var(--radius) + 6px)',
          '2xl': 'calc(var(--radius) + 9px)',
          '3xl': 'calc(var(--radius) + 12px)',
          full: '9999px',
        },
      },
      plugins: [],
    } satisfies Config;
    ```

  </Step>

  <Step>
    ### Add tailwind animations

    Extend the `keyframes` and `animation` objects in your `theme` object.

    ```ts title="tailwind.config.ts"
    import type { Config } from 'tailwindcss';

    export default {
      darkMode: 'class',
      content: [
        '...',
      ],
      theme: {
        // ...
        extend: {
          // ...
          // [!code highlight:33]
          keyframes: {
            'accordion-down': {
              from: { height: '0' },
              to: { height: 'var(--radix-accordion-content-height)' },
            },
            'accordion-up': {
              from: { height: 'var(--radix-accordion-content-height)' },
              to: { height: '0' },
            },
            'caret-blink': {
              '0%,70%,100%': { opacity: '1' },
              '20%,50%': { opacity: '0' },
            },
            'collapsible-down': {
              from: { height: '0' },
              to: {
                height: 'var(--radix-collapsible-content-height)',
              },
            },
            'collapsible-up': {
              from: {
                height: 'var(--radix-collapsible-content-height)',
              },
              to: { height: '0' },
            },
          },
          animation: {
            'accordion-down': 'accordion-down 0.2s ease-out',
            'accordion-up': 'accordion-up 0.2s ease-out',
            'caret-blink': 'caret-blink 1.25s ease-out infinite',
            'collapsible-down': 'collapsible-down 0.15s ease-out',
            'collapsible-up': 'collapsible-up 0.15s ease-out',
          },
        },
      },
    } satisfies Config;
    ```

  </Step>

  <Step>
    ### Add tailwind plugins
    
    Add the `tailwindcss-animate` plugin to your Tailwind CSS config.

    ```ts title="tailwind.config.ts"
    import type { Config } from 'tailwindcss';
    import animate from 'tailwindcss-animate'; // [!code highlight]

    export default {
      // ...
      plugins: [animate], // [!code highlight]
    } satisfies Config;
    ```

  </Step>

    ### Add utilities
    
    Create the `cn()` utility function.

    ```ts title="app/utils/cn.ts"
    import type { ClassValue } from 'clsx';
    import { clsx } from 'clsx';
    import { twMerge } from 'tailwind-merge';

    export const cn = (...inputs: ClassValue[]) => twMerge(clsx(inputs));
    ```
