## Usage

```ts
import {
  ResizableHandle,
  ResizablePanel,
  ResizablePanelGroup,
} from '~/components/ui/resizable';
```

```tsx
<ResizablePanelGroup direction='horizontal'>
  <ResizablePanel>One</ResizablePanel>
  <ResizableHandle />
  <ResizablePanel>Two</ResizablePanel>
</ResizablePanelGroup>
```

## Examples

### Vertical

Use the `direction` prop to set the direction of the resizable panels.

<ComponentPreview name='resizable-vertical'>
```json doc-gen:file
{
  "file": "./src/components/demos/Resizable/ResizableVertical.tsx",
  "codeblock": true
}
```
</ComponentPreview>

```tsx
import {
  ResizableHandle,
  ResizablePanel,
  ResizablePanelGroup,
} from '~/components/ui/resizable';

export const Example = () => {
  return (
    // [!code highlight:2]
    <ResizablePanelGroup direction='vertical'>
      <ResizablePanel>One</ResizablePanel>
      <ResizableHandle />
      <ResizablePanel>Two</ResizablePanel>
    </ResizablePanelGroup>
  );
};
```

### Handle

You can set or hide the handle by using the `withHandle` prop on the `<ResizableHandle />` component.

<ComponentPreview name='resizable-handle'>
```json doc-gen:file
{
  "file": "./src/components/demos/Resizable/ResizableHandle.tsx",
  "codeblock": true
}
```
</ComponentPreview>

```tsx
import {
  ResizableHandle,
  ResizablePanel,
  ResizablePanelGroup,
} from '~/components/ui/resizable';

export const Example = () => {
  return (
    <ResizablePanelGroup direction='horizontal'>
      <ResizablePanel>One</ResizablePanel>
      <ResizableHandle withHandle /> // [!code highlight]
      <ResizablePanel>Two</ResizablePanel>
    </ResizablePanelGroup>
  );
};
```
