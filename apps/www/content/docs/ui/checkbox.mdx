  <Tab value='Manual'>

        ```

      <Step>
        ### Copy-paste the component

        Copy and paste the component code in a `.tsx` file.

        ```

      </Step>

```ts
import { Checkbox } from '~/components/ui/checkbox';
```

<Checkbox />
