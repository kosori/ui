## Usage

```ts
import { Avatar, AvatarFallback, AvatarImage } from '~/components/ui/avatar';
```

```tsx
<Avatar>
  <AvatarImage src='https://github.com/codingcodax.png' />
  <AvatarFallback>CC</AvatarFallback>
</Avatar>
```
