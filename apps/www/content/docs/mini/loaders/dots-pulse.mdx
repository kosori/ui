## Installation

<Steps>
    
  <Step>
    ### Add the keyframe
    
    Add the keyframe to the `keyframes` object in your `tailwind.config.js` file.

    ```ts title="tailwind.config.ts"
    import type { Config } from 'tailwindcss';

    const config: Config = {
      // ...
      theme: {
        // ...
        extend: {
          // ...
          keyframes: {
            // [!code highlight:6]
            flash: {
              '0%': { opacity: '0.2' },
              '20%': { opacity: '1' },
              '100%': { opacity: '0.2' },
            },
          },
        },
      },
    };

    export default config;
    ```

  </Step>
    
  <Step>
    ### Add the animation
    
    Add the animation to the `animation` object in your `tailwind.config.js` file.

    ```ts title="tailwind.config.ts"
    import type { Config } from 'tailwindcss';

    const config: Config = {
      // ...
      theme: {
        // ...
        extend: {
          // ...
          animation: {
            flash: 'flash 1.4s infinite linear', // [!code highlight]
          },
        },
      },
    };

    export default config;
    ```

  </Step>

</Steps>
